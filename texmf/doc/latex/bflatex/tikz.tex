\documentclass{standalone}
\usepackage{tikz}
\usepackage{pgfplots}
\usepackage[firafull]{addfonts}

\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
\begin{axis}[
  width=10cm,
  height=7cm,
  xlabel={$x$},
  ylabel={$f(x)$},
  axis lines=center,
  xmin=-3.5, xmax=3.5,
  ymin=-1.5, ymax=1.5,
  xtick={-3,-2,-1,0,1,2,3},
  ytick={-1,-0.5,0,0.5,1},
  grid=major,
  grid style={line width=0.2pt, draw=gray!30},
  tick style={color=black},
  every axis plot/.append style={line width=1.2pt},
]
\addplot[blue, domain=-3.5:3.5, samples=200] {sin(deg(x))};
\addlegendentry{$\sin(x)$}
\addplot[red, domain=-3.5:3.5, samples=200] {cos(deg(x))};
\addlegendentry{$\cos(x)$}
\end{axis}
\end{tikzpicture}
\end{document}
